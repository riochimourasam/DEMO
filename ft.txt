    <script>
        // Slider automatique pour le hero
        let currentSlide = 0;
        const slides = 4; // Nombre de slides
        
        function updateSlideIndicators() {
            const indicators = document.querySelectorAll('.max-w-7xl.mx-auto.px-8.pb-8 button');
            indicators.forEach((indicator, index) => {
                if (index === currentSlide) {
                    indicator.className = 'w-4 h-4 rounded-full bg-white';
                } else {
                    indicator.className = 'w-4 h-4 rounded-full bg-white/40';
                }
            });
        }
        
        function nextSlide() {
            currentSlide = (currentSlide + 1) % slides;
            updateSlideIndicators();
        }
        
        // Change de slide toutes les 5 secondes
        setInterval(nextSlide, 5000);
        
        // Gestion des clics sur les indicateurs
        document.querySelectorAll('.max-w-7xl.mx-auto.px-8.pb-8 button').forEach((button, index) => {
            button.addEventListener('click', () => {
                currentSlide = index;
                updateSlideIndicators();
            });
        });
    </script>
